Using HTML attributes only



                
<div class="colorPicker" for="img" trigger="btn" 
onchange="btn.style.backgroundColor = this.value"></div>

Using HTML attributes and JS event


                
<div class="colorPicker" 
for="img" trigger="btn" 
onchange="onColorPickerChanged"></div>

function onColorPickerChanged(e) {
    btn.style.backgroundColor = e.value;
}

Using JavaScript


                
<div class="colorPicker" for="img" ></div>

btn.addEventListener('click', function () {
    var colorPicker = document.querySelector('.colorPicker');
    colorPicker.addEventListener("onchange", function (e) {
        btn.style.backgroundColor = e.detail.value;
    });
    colorPicker.addEventListener("onColorPickerEnabled", function (e) {
        btn.innerHTML = "onColorPickerEnabled";
    });
    colorPicker.addEventListener("onColorPickerDisabled", function (e) {
        btn.innerHTML = "onColorPickerDisabled";
    });
    colorPicker.enableColorPicker();
});
